1. 概述

本文档设计 Android APK 在 OpenHarmony (OH) 系统上的安装机制。文档首先分别分析 Android 和 OH 两个系统中包管理与安装守护进程的实现原理， 然后基于对两个系统的深入理解，给出将 Android APK 安装到 OH 系统并正确启动运行的完整设计方案。

文档结构

章节内容目的
第 2 章Android installd + PMS 实现原理理解 Android 端接口与流程，明确需要适配的接口

第 4 章APK 安装到 OH 的适配方案桥接两端差异，给出完整设计

前置依赖

  • appspawn-x（已设计）— Android 应用进程孵化器，提供 ART VM + 预加载 + fork
  • 适配层框架（已实现）— ActivityManagerAdapter、IntentWantConverter 等
  • BroadcastEventConverter（已实现）— OH CommonEvent ↔ Android Broadcast 映射

2. Android 系统中 installd 与 PMS 的实现原理

2.1 Android 包管理架构总览

Android 的包管理由三层组成：客户端 API 层（App 使用）、系统服务层（PMS）、Native 守护进程层（installd）。

2.5 包查询端到端流程

// App 调用
PackageInfo info = pm.getPackageInfo("com.example.app", PackageManager.GET_ACTIVITIES);

// 调用链：
App.getPackageInfo()
  → ApplicationPackageManager.getPackageInfo()        // 客户端代理
    → IPackageManager.Stub.Proxy.getPackageInfo()     // Binder 序列化
      → [Binder IPC 传输]
        → PackageManagerService.getPackageInfo()      // 服务端入口
          → Computer.snapshotComputer()               // 获取一致性快照
            → ComputerEngine.getPackageInfo()         // 查询实现
              → mPackages.get(packageName)            // HashMap 查找
              → generatePackageInfo(ps, flags)        // 按 flags 构建返回值

快照模式 (Snapshot Pattern)：PMS 使用 ComputerEngine 的只读快照处理查询请求。 每次查询调用 snapshotComputer() 获取不可变的数据副本，避免加锁，保证并发安全。

3. OH 系统中 installd 与 BMS 的实现原理

3.1 OH 包管理架构总览

OH 的包管理架构与 Android 类似，也分三层：客户端 API 层、系统服务层（BMS）、Native 守护进程层（installd）。 但 OH 使用 HiRPC/Binder 替代 Android Binder，使用 HAP 替代 APK，使用 module.json 替代 AndroidManifest.xml。

4.2 方案选型与整体架构

方案描述优点缺点评估
A: APK → HAP 转换 安装时将 APK 重打包为 HAP 格式 BMS 无需修改 classes.dex ≠ .abc，转换无意义；资源格式不兼容 ❌ 不可行
B: 并行 PMS 独立运行 Android PMS 服务 完全兼容 Android 需移植 PMS（10万+行）；双系统不一致；桌面不可见 ❌ 过重
C: BMS 扩展 + APK 解析插件 在 BMS 安装流程中增加 APK 分支 最小修改；统一管理；BMS 查询正常 需实现 AXML 解析器；BMS 需少量修改 ✅ 选用
D: 安装代理 独立服务安装后向 BMS 注入记录 BMS 不修改 BMS 无注入接口；一致性风险 ⚠️ 备选
选定方案 C：BMS 扩展 + APK 解析插件。在 BMS 的 BundleInstaller 中增加 APK 检测分支， 当 .apk 文件进入安装流程时，调用 APK 解析插件（而非 HAP 解析器）解析 AndroidManifest.xml 并生成 InnerBundleInfo。 同时扩展 OH installd 以支持 APK 特有的操作（dex2oat、Native 库提取、Android 数据目录创建）。

// installd_host_impl_android.cpp — Android APK 专用 installd 扩展

ErrCode InstalldHostImpl::CreateAndroidDataDir(
        const std::string& bundleName, int32_t uid, int32_t gid,
        int32_t targetSdkVersion) {
    std::string baseDir = "/data/app/el2/0/android/" + bundleName;

    // 创建 Android 风格子目录
    const std::vector<std::pair<std::string, mode_t>> dirs = {
        {"",              0771},
        {"/cache",        0771},
        {"/code_cache",   0771},
        {"/databases",    0771},
        {"/files",        0771},
        {"/shared_prefs", 0771},
        {"/lib",          0755},
    };

    for (const auto& [sub, mode] : dirs) {
        std::string path = baseDir + sub;
        OHOS::ForceCreateDirectory(path);
        chmod(path.c_str(), mode);
        chown(path.c_str(), uid, gid);
    }

    // 外部存储目录
    std::string extDir = "/data/app/el2/0/android_ext/" + bundleName;
    OHOS::ForceCreateDirectory(extDir);
    chown(extDir.c_str(), uid, /* sdcard_rw gid */);

    return ERR_OK;
}

ErrCode InstalldHostImpl::DexOptAndroid(
        const std::string& apkPath, const std::string& oatDir,
        int32_t uid, const std::string& isa,
        const std::string& compilerFilter) {
    // 构建 dex2oat 参数
    std::string oatFile = oatDir + "/base.odex";
    std::vector<std::string> args = {
        "/system/android/bin/dex2oat",
        "--dex-file=" + apkPath,
        "--oat-file=" + oatFile,
        "--instruction-set=" + isa,
        "--compiler-filter=" + compilerFilter,
        "--boot-image=/system/android/framework/boot.art",
        "--android-root=/system/android",
    };

    pid_t pid = fork();
    if (pid == 0) {
        setuid(uid);
        // 构建 argv
        std::vector<const char*> argv;
        for (auto& a : args) argv.push_back(a.c_str());
        argv.push_back(nullptr);
        execv(argv[0], const_cast<char**>(argv.data()));
        _exit(127);
    }
    int status;
    waitpid(pid, &status, 0);
    return (WIFEXITED(status) && WEXITSTATUS(status) == 0) ? ERR_OK : ERR_APPEXECFWK_INSTALLD_DEXOPT_FAILED;
}

ErrCode BundleInstaller::ProcessApkInstall(const std::string& apkPath, ...) {
    // 1. 解析 AndroidManifest.xml (AXML)
    ApkManifestParser::ManifestData manifest;
    if (!ApkManifestParser::Parse(apkPath, manifest)) {
        return ERR_APPEXECFWK_INSTALL_INVALID_APK;
    }

    // 2. APK 签名验证
    auto installd = DelayedSingleton<InstalldClient>::GetInstance();
    installd->VerifyApkSignature(apkPath, "");

    // 3. 版本检查（如果是更新）
    CheckVersionForUpdate(manifest.packageName, manifest.versionCode);

    // 4. 复制 APK 到安装目录
    std::string installDir = "/data/app/android/" + manifest.packageName;
    installd->Mkdir(installDir, 0755, uid, gid);
    installd->CopyFile(apkPath, installDir + "/base.apk");

    // 5. 提取 Native 库
    installd->ExtractApkNativeLibs(apkPath, installDir + "/lib", primaryAbi);

    // 6. DEX 优化
    installd->DexOptAndroid(installDir + "/base.apk",
                            installDir + "/oat/arm64",
                            uid, "arm64", "speed");

    // 7. 创建 Android 数据目录
    installd->CreateAndroidDataDir(manifest.packageName, uid, gid,
                                    manifest.targetSdkVersion);

    // 8. 构建 InnerBundleInfo 并注册到 BMS
    InnerBundleInfo bundleInfo = ApkBundleParser::Build(manifest, installDir);
    bundleDataMgr_->AddInnerBundleInfo(manifest.packageName, bundleInfo);

    // 9. 发送安装事件
    SendCommonEvent(CommonEventConstants::BUNDLE_ADDED, manifest.packageName);

    return ERR_OK;
}

// apk_bundle_parser.cpp — 核心映射函数
InnerBundleInfo ApkBundleParser::Build(
        const ApkManifestParser::ManifestData& manifest,
        const std::string& installDir) {

    InnerBundleInfo bundleInfo;

    // 1. ApplicationInfo
    ApplicationInfo appInfo;
    appInfo.name = manifest.appClassName.empty()
                   ? manifest.packageName : manifest.appClassName;
    appInfo.bundleName = manifest.packageName;
    appInfo.versionCode = manifest.versionCode;
    appInfo.versionName = manifest.versionName;
    appInfo.codePath = installDir + "/base.apk";
    appInfo.dataDir = "/data/app/el2/0/android/" + manifest.packageName;
    appInfo.isSystemApp = false;
    appInfo.bundleType = BundleType::APP_ANDROID;
    appInfo.process = manifest.packageName;
    appInfo.apiTargetVersion = manifest.targetSdkVersion;
    bundleInfo.SetApplicationInfo(appInfo);

    // 2. HapModuleInfo（单 APK = 单模块 "entry"）
    HapModuleInfo moduleInfo;
    moduleInfo.name = "entry";
    moduleInfo.moduleName = "entry";
    moduleInfo.bundleName = manifest.packageName;
    moduleInfo.hapPath = installDir + "/base.apk";
    moduleInfo.nativeLibraryPath = installDir + "/lib/arm64-v8a";
    moduleInfo.type = ModuleType::ENTRY;

    // 3. Activities → AbilityInfos
    for (const auto& activity : manifest.activities) {
        AbilityInfo abilityInfo;
        abilityInfo.name = MapClassToAbilityName(activity.name);
        abilityInfo.bundleName = manifest.packageName;
        abilityInfo.moduleName = "entry";
        abilityInfo.type = AbilityType::PAGE;
        abilityInfo.launchMode = MapLaunchMode(activity.launchMode);
        abilityInfo.orientation = MapOrientation(activity.screenOrientation);
        abilityInfo.visible = activity.exported;
        abilityInfo.codePath = installDir + "/base.apk";
        abilityInfo.srcLanguage = "java";

        // Intent-filter → Skills
        for (const auto& filter : activity.intentFilters) {
            Skill skill;
            for (const auto& action : filter.actions)
                skill.actions.push_back(MapAction(action));
            for (const auto& category : filter.categories)
                skill.entities.push_back(MapCategory(category));
            for (const auto& data : filter.dataSpecs) {
                SkillUri uri;
                uri.scheme = data.scheme; uri.host = data.host;
                uri.path = data.path;     uri.type = data.type;
                skill.uris.push_back(uri);
            }
            abilityInfo.skills.push_back(skill);
        }
        moduleInfo.abilityInfos.push_back(abilityInfo);
        bundleInfo.InsertAbilityInfo(abilityInfo);
    }

    // 4. Services → ExtensionAbilityInfos (SERVICE)
    for (const auto& service : manifest.services) {
        ExtensionAbilityInfo extInfo;
        extInfo.name = service.name;
        extInfo.bundleName = manifest.packageName;
        extInfo.type = ExtensionAbilityType::SERVICE;
        extInfo.visible = service.exported;
        bundleInfo.InsertExtensionInfo(extInfo);
    }

    // 5. BroadcastReceivers → ExtensionAbilityInfos (STATICSUBSCRIBER)
    for (const auto& receiver : manifest.receivers) {
        ExtensionAbilityInfo extInfo;
        extInfo.name = receiver.name;
        extInfo.bundleName = manifest.packageName;
        extInfo.type = ExtensionAbilityType::STATICSUBSCRIBER;
        extInfo.visible = receiver.exported;
        bundleInfo.InsertExtensionInfo(extInfo);
    }

    // 6. ContentProviders → ExtensionAbilityInfos (DATASHARE)
    for (const auto& provider : manifest.providers) {
        ExtensionAbilityInfo extInfo;
        extInfo.name = provider.name;
        extInfo.bundleName = manifest.packageName;
        extInfo.type = ExtensionAbilityType::DATASHARE;
        extInfo.visible = provider.exported;
        extInfo.uri = "datashare:///" + provider.authorities;
        bundleInfo.InsertExtensionInfo(extInfo);
    }

    // 7. 权限映射
    for (const auto& perm : manifest.usesPermissions) {
        RequestPermission reqPerm;
        reqPerm.name = PermissionMapper::MapToOH(perm);
        reqPerm.moduleName = "entry";
        bundleInfo.AddRequestPermission(reqPerm);
    }

    bundleInfo.InsertModuleInfo(moduleInfo);
    return bundleInfo;
}

4.8.2 Native 库 ABI 选择策略

// ABI 优先级（ARM64 设备）
static const std::vector<std::string> kSupportedAbis = {
    "arm64-v8a",      // 优先：64 位 ARM
    "armeabi-v7a",    // 回退：32 位 ARM
    "armeabi"         // 遗留：ARMv5
};

std::string SelectPrimaryAbi(const std::string& apkPath) {
    ZipArchive zip(apkPath);
    for (const auto& abi : kSupportedAbis) {
        std::string prefix = "lib/" + abi + "/";
        if (zip.HasEntriesWithPrefix(prefix)) {
            return abi;
        }
    }
    return "";  // 无 Native 库
}

4.8.3 Native 库提取过程

  1. 扫描 APK ZIP 中 lib/{abi}/ 目录
  2. 选择设备支持的最佳 ABI（按 kSupportedAbis 优先级）
  3. 提取所有 .so 文件到 /data/app/android/{pkg}/lib/{abi}/
  4. 设置文件权限（owner=uid, mode=0755）
  5. 如果 AndroidManifest.xml 中 extractNativeLibs=false，则不提取，运行时直接从 APK 中加载（需 APK 页对齐）

4.10.1 PackageInfoBuilder — BundleInfo → PackageInfo 转换

// PackageInfoBuilder.java — 将 OH BundleInfo 转换为 Android PackageInfo
public class PackageInfoBuilder {
    public static PackageInfo fromBundleInfo(String bundleInfoJson) {
        PackageInfo pi = new PackageInfo();
        JSONObject json = new JSONObject(bundleInfoJson);

        pi.packageName = json.getString("name");
        pi.versionCode = json.getInt("versionCode");
        pi.versionName = json.getString("versionName");

        // ApplicationInfo
        pi.applicationInfo = new ApplicationInfo();
        pi.applicationInfo.packageName = pi.packageName;
        pi.applicationInfo.sourceDir = "/data/app/android/" + pi.packageName + "/base.apk";
        pi.applicationInfo.dataDir = "/data/data/" + pi.packageName;
        pi.applicationInfo.uid = json.optInt("uid", -1);
        pi.applicationInfo.targetSdkVersion = json.optInt("maxSdkVersion", 33);

        // AbilityInfos → ActivityInfo[] (还原 Ability 后缀)
        JSONArray abilities = json.optJSONArray("abilityInfos");
        if (abilities != null) {
            pi.activities = new ActivityInfo[abilities.length()];
            for (int i = 0; i < abilities.length(); i++) {
                JSONObject ab = abilities.getJSONObject(i);
                ActivityInfo ai = new ActivityInfo();
                String abilityName = ab.getString("name");
                ai.name = IntentWantConverter.abilityNameToClassName(
                    pi.packageName, abilityName);
                ai.packageName = pi.packageName;
                ai.exported = ab.optBoolean("visible", false);
                pi.activities[i] = ai;
            }
        }

        // 权限反向映射
        JSONArray perms = json.optJSONArray("reqPermissions");
        if (perms != null) {
            pi.requestedPermissions = new String[perms.length()];
            for (int i = 0; i < perms.length(); i++) {
                pi.requestedPermissions[i] =
                    PermissionMapper.mapToAndroid(perms.getString(i));
            }
        }
        return pi;
    }
}
